table inet filter {
	chain input {
		type filter hook input priority filter; policy drop;
		iifname "lo" accept
		ct state established,related accept
		iifname "eth1" icmp type echo-request accept
		iifname "eth0" icmp type echo-request accept
		iifname "eth0" ip saddr 192.168.56.0/24 tcp dport 22 accept
	}

	chain forward {
		type filter hook forward priority filter; policy drop;
		ct state established,related accept
		iifname "eth0" oifname "eth1" drop
		iifname "eth1" oifname "eth0" udp dport { 53, 123 } accept
		iifname "eth1" oifname "eth0" tcp dport 443 accept
	}
}

	chain postrouting {

		oifname "eth0" ip saddr 10.20.0.0/24 masquerade
	}
